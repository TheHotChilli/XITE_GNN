20 def count_pair_occurrences(df_AU_data, labels, eps_activity=1.0, comparator=np.logical_and):
21  """!
22  Computes the absolute co-occurence frequencies for each AU pairs, given the AU-Data of one subject.
23  The co-occurence frequencies is defined as:
24  n_{AU_i \and AU_j} = \# frames where AU_i and AU_j co-occure
25 
26  @param df_AU_features Pandas Dataframe containing label information and AU classification or reggression value per frame; shape = (nof_frames, nof_AU + 1);
27  @param eps_activity only values >= eps_activity will be counted; set to 1 if using classification results; set in [0,5] if using regression results
28  @retval df_freq_abs Pandas Multiindex Dataframe for storing the results; shape = (labels x nof_AU, nof_AU x 2)
29  """
30 
31  assert(0 <= eps_activity <= 5)
32 
33  labels_valid = [0,1,2,3,4,5,6,-1,-2,-3,-4,-5,-6]
34  AU_names = list(df_AU_data.columns)
35 
36  rowidx = pd.MultiIndex.from_product([labels_valid, AU_names], names=["label", "AU"])
37  colidx = pd.Index(AU_names, name="AU")
38  df_counts = pd.DataFrame(0, index=rowidx, columns=colidx)
39 
40  # count occurences
41  for label in labels_valid:
42  for AUi in AU_names:
43  for AUj in AU_names:
44  df_counts.loc[(label, AUi), AUj] = df_AU_data[(labels == label) \
45  & comparator((df_AU_data[AUi] >= eps_activity), (df_AU_data[AUj] >= eps_activity)).values] \
46  .iloc[:,0].count()
47  return df_counts

50 def _subject_frequency_analysis(subject_id, eps_conf=None):
51  """!
52  Wrapper function. Performs the frequency analysis on a single subject with provided subject_id.
53 
54  @param subject_id (int) ID of the subject to be analyzed.
55  @param eps_conf (float, optional) Confidence threshold for filtering out inconfident data.
56  @retval df_list list of pandas DataFrames and a pandas Series. Four DataFrames with the count of AUs appearing
57  with each other under different logical operations and a Series with the total number of frames
58  per label.
59  """
60  # init
61  labels_valid = [0,1,2,3,4,5,6,-1,-2,-3,-4,-5,-6]
62  dataset = OpenFace_Raw_Dataset()
63  labels = dataset.load_labels(subject_id)
64  AUc_data = dataset.load_data(subject_id, target_channels="AUc")
65  AUr_data = dataset.load_data(subject_id, target_channels="AUr")
66  nof_frames_per_label = pd.Series(0, index=labels_valid)
67  # optional: filter out inconfident data
68  if eps_conf:
69  assert(0.0 < eps_conf <= 1.0)
70  confidence = dataset.load_data(subject_id, target_channels="confidence")
71  AUc_data = AUc_data[confidence>eps_conf]
72  AUr_data = AUr_data[confidence>eps_conf]
73  labels = labels[labels>eps_conf]
74  # count occurences
75  df_counts_AUc_and = count_pair_occurrences(AUc_data, labels, comparator=np.logical_and)
76  df_counts_AUc_or = count_pair_occurrences(AUc_data, labels, comparator=np.logical_or)
77  df_counts_AUr_and = count_pair_occurrences(AUr_data, labels, comparator=np.logical_and)
78  df_counts_AUr_or = count_pair_occurrences(AUr_data, labels, comparator=np.logical_or)
79  # count total nof frames per label
80  for label in labels_valid:
81  nof_frames_per_label[label] += len(labels[labels == label])
82  print(f"Frequency analysis finished for S{subject_id}")
83 
84  return [df_counts_AUc_and, df_counts_AUc_or, df_counts_AUr_and, df_counts_AUr_or, nof_frames_per_label]
