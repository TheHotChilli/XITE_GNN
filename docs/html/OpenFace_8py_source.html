160  def _transform_labels(self):
161  """!

164  """

292  def _load_data(self, use_labels=None):
293  df = pd.read_csv(self.root_path)
294  if use_labels:
295  df = df[df["label"].isin(use_labels)]
296  # reset index
297  end_idxs = np.diff(df["slice_id"]).nonzero()[0]
298  start_idxs = end_idxs + 1
299  end_idxs = np.append(end_idxs, [len(df)-1])
300  start_idxs = np.concatenate([[0], start_idxs])
301  new_idx = np.ndarray(len(df))
302  for i in range(len(start_idxs)):
303  new_idx[start_idxs[i]:end_idxs[i]] = i
304  df.set_index(new_idx)
