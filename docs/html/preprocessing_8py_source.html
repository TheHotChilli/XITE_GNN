254  def _compute_slices_info(self, labels, f_sampling):
255  """
256  Computes information about where each slice starts.
257 
258  @param[in] labels np.array; label info for each frame
259  @param[in] f_sampling int; sampling frequency
260  @param[out] slices_start_idxs list; list of arrays, each array contains start idxs for one label
261  @param[out] slices_labels list;
262  """
263 
264  # interval information
265  start_idxs, end_idxs, interval_labels, durations = self._compute_intervals_info(labels) # get info about all intervals
266  pre_interval_labels = np.concatenate(([np.nan], interval_labels[0:-1]))
267  pre_durations = np.concatenate(([np.nan], durations[0:-1]))
268  post_interval_labels = np.concatenate((interval_labels[1::], [np.nan]))
269  post_durations = np.concatenate((durations[1::], [np.nan]))
270 
271  # containers for accumulation
272  slices_start_idxs = [] #list of arrays, each array represents the global slice start idxs for one label
273  slices_labels = [] #list of labels, each entry maps array in start_idxs to a label
274 
275  # check which intervals
276  for label in self.labels_pain:
277  pain_group = self._get_label_group(label)
278  # find pain intervals that match (check if label matches, pre and post interval are baseline, and if durations ok)
279  is_pain_intervals = (
280  (interval_labels == label) # interval has pain label?
281  & (np.isin(pre_interval_labels, self.labels_base + [0])) # pre interval has baseline label?
282  & (np.isin(post_interval_labels, self.labels_base + [0])) # post interval has baseline label?
283  & (pre_durations >= self.pre_interval_min_lengths[pain_group]) # pre (base)interval is long enough
284  & (durations >= self.interval_min_lengths[pain_group]*f_sampling-2) # interval is long enough
285  # & (post_durations >= self.slice_shifts["B"+pain_group]*f_sampling # post (base)interval is long enough for shifted baseline slice
286  # + self.slice_lengths["B"+pain_group]*f_sampling) # B_duration >= B_shift + B_len
287  )
288  pain_interval_ids = is_pain_intervals.nonzero()[0]
289  # assure pain slice is not interlapping with next stimuli interval
290  nof_interlapping_frames = self.slice_shifts[pain_group]*f_sampling +\
291  + self.slice_lengths[pain_group]*f_sampling - durations[pain_interval_ids] # interlapping = shift + slice_len - duration
292  pain_interval_ids = pain_interval_ids[nof_interlapping_frames <= post_durations[pain_interval_ids]]
293 
294  #base_interval_ids = pain_interval_ids + 1
295  is_base_intervals = (interval_labels == label*100) & (pre_interval_labels == label)
296  base_interval_ids = is_base_intervals.nonzero()[0]
297  # check that baseline interval is long enough for shifted baseline slice
298  base_interval_ids = base_interval_ids[durations[base_interval_ids] >= \
299  self.slice_shifts["B"+pain_group]*f_sampling + self.slice_lengths["B"+pain_group]*f_sampling] # B_duration >= B_shift + B_len
300 
301  # Compute actual slice info
302  pain_slice_start_idxs = start_idxs[pain_interval_ids] + self.slice_shifts[pain_group]*f_sampling
303  base_slice_start_idxs = start_idxs[base_interval_ids] + self.slice_shifts["B"+pain_group]*f_sampling
304  # Append info arrays of pain slices and base slices to list
305  slices_start_idxs += list(pain_slice_start_idxs)
306  slices_labels += [label for i in range(len(pain_slice_start_idxs))]
307  slices_start_idxs += list(base_slice_start_idxs)
308  slices_labels += [label*100 for i in range(len(base_slice_start_idxs))]
309 
310  return slices_start_idxs, slices_labels

313  def _compute_intervals_info(self, labels):
314  """!
315  Computes the information about all intervals (pain, baseline and invalid labels). Returns for each interval
316  information about start_idx, end_idx, interval_label and duration.
317  """
318  is_interval_start = np.concatenate(([True], labels[:-1] != labels[1:]), axis=0)
319  start_idxs = is_interval_start.nonzero()[0]
320  end_idxs = np.roll(is_interval_start, -1).nonzero()[0]
321  durations = end_idxs - start_idxs + 1
322  interval_labels = labels[start_idxs]
323  # set label of baseline intervals that follow a pain interval to 100*label
324  pre_interval_labels = np.concatenate(([np.nan], interval_labels[0:-1]))
325  is_baseline_interval = (interval_labels == 0 &
326  np.isin(pre_interval_labels, self.labels_pain))
327  interval_labels[is_baseline_interval] = pre_interval_labels[is_baseline_interval]*100
328  return start_idxs, end_idxs, interval_labels, durations

341  @retval features (pd.DataFrame) Dataframe with feature data of all slices. Columns represent channels and their features.
342  Channel index is a multiindex where index-level 0 represents the channel and index-level 1 represents the corresponding features.

398  def _extract_features_from_slice(self, data, sample_rate):
399  """!
400  This function takes the DataFrame of a single slice and the sample_rate at which the data was
401  recorded and extractes several features for each data channel (data column). Feature extraction
402  methods are specified via config file. Returns a list of feature values.
403 
404  @param data (pd.DataFrame) DataFrame representing the data of a single slice.
405  @param sample_rate (int or float) Sample rate at which data was recorded.
406 
407  @retval feature (list) List of features. Each entry is one feature.
408  """
409  # init empty list for aggregation of all slice features
410  features = []
411  # loop each channel data channel --> extract features for each channel seperate
412  for channel in data.columns:
413  # apply feature extraction on signal
414  if channel in self.channels_feature_extraction:
415  for func_name in self.feature_extraction_settings[channel]:
416  func = getattr(feature_extraction_functions, func_name)
417  try:
418  features.append(func(data[channel].to_numpy(), sample_rate=sample_rate)) #try calling func with opt param
419  except:
420  features.append(func(data[channel].to_numpy())) # call func without opt param
421  # apply feature extraction on 1st derivative
422  if channel in self.channels_feature_extraction_d1:
423  derivative = feature_extraction_functions.compute_derivative(data[channel].to_numpy())
424  for func_name in self.feature_extraction_settings_d1[channel]:
425  func = getattr(feature_extraction_functions, func_name)
426  try:
427  features.append(func(derivative, sampling_rate=sample_rate))
428  except:
429  features.append(func(derivative))
430  # apply feature extraction on 2nd derivative
431  if channel in self.channels_feature_extraction_d2:
432  derivative = feature_extraction_functions.compute_derivative(\
433  feature_extraction_functions.compute_derivative(data[channel].to_numpy()))
434  for func_name in self.feature_extraction_settings_d2[channel]:
435  func = getattr(feature_extraction_functions, func_name)
436  try:
437  features.append(func(derivative, sample_rate=sample_rate))
438  except:
439  features.append(func(derivative))
440  return features

443  def _repair_features(self,features, strategy="mean"):
444  """!
445  Replaces or removes NaN, inf, or -inf entries in the input features data.
446 
447  @param features (pd.DataFrame) Feature dataset of all samples/slices.
448  @param strategy (str or float) The strategy to repair corrupt entries in features. Valid options are "mean", "delete",
449  or a float/integer that indicates a replacement value.
450  """
451 
452  features.replace([np.inf, -np.inf], np.nan, inplace=True)
453 
454  # Delete all rows where an corrupt entrie appears
455  if strategy == "delete":
456  return features.dropna(axis=0, how="any", inplace=True)
457 
458  # Try to replace corrupt entries with the mean value of the column/feature
459  # of all non corrupt entries with matching label
460  elif strategy == "mean":
461  for label in np.unique(features.index.get_level_values("label")):
462  data_label = features.loc[(slice(None),slice(None),label,slice(None),slice(None)),:]
463  features.loc[(slice(None),slice(None),label,slice(None),slice(None)),:] = data_label.fillna(value=data_label.mean(), axis=0)
464  features.dropna(axis=0, how="any", inplace=True)
465  return features
466 
467  elif type(strategy) == int or type(strategy) == float:
468  features.fillna(value=strategy, axis=0, inplace=True)
469  return features
470 
471  else:
472  raise(ValueError(f"Invalid strategy '{strategy}'! Supported strategies are mean, delete or\
473  replacement with a providede value."))
