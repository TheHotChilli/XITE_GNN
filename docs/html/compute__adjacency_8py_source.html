14 def compute_rel_AU_freqs_uncond(df_counts_and, nof_frames):
15  """!
16  Computes for each AU pair the relative co-occurance frequency.
17  P(AU_i, AU_j) = n_{AU_i \land AU_j} / n_{total frames with label}
18 
19  @param df_AU_abs_freq Pandas Multiindex Dataframe with absolute AU Co-Occurence frequencies; shape = (labels x nof_AU, nof_AU)
20  @param nof_frames Pandas Series with information about total number of frames per label; shape = (nof_labels, 1)
21  @retval df_AU_rel_freq Pandas Multiindex Dataframe with relative AU Co-Occurence frequencies; shape = (labels x nof_AU, nof_AU)
22  """
23 
24  # allocate memory for results
25  AU_names = list(df_counts_and.columns)
26  df_rel_freqs = pd.DataFrame(0, index=AU_names, columns=AU_names)
27 
28  # compute (unconditional) probabibility of Co-Occurrance
29  for AUi in AU_names:
30  for AUj in AU_names:
31  df_rel_freqs.loc[AUi, AUj] = df_counts_and.loc[AUi,AUj] / nof_frames
32 
33  return df_rel_freqs

36 def compute_rel_AU_freqs_cond(df_counts_and):
37  """!
38  Takes as input the absolute AU Co-Occurence frequencies stored in a Dataframe df_AU_freq and computes
39  for all AU combinations the conditional probability for Co-Occurence:
40  P(AU_i | AU_j) = n_{AU_i \land AU_j} / n_{AU_j}
41 
42  @param df_counts_and Pandas Multiindex Dataframe with absolute AU Co-Occurence frequencies; shape = (nof_labels x nof_AU, nof_AU)
43  @retval df_rel_freqs Pandas Multiindex Dataframe with conditional relative AU Co-Occurence frequencies; shape = (labels x nof_AU, nof_AU)
44  """
45  # allocate memory for results
46  AU_names = list(df_counts_and)
47  df_rel_freqs = pd.DataFrame(0, index=AU_names, columns=AU_names)
48 
49  # Compute conditional probability of Co-Occurance of AU Combinations:
50  for AUi in AU_names:
51  for AUj in AU_names:
52  df_rel_freqs.loc[AUi, AUj] = df_counts_and.loc[AUi, AUj] / df_counts_and.loc[AUj, AUj]
53 
54  # replace nan's; can have nan results if n_AUjAUj = 0
55  df_rel_freqs.fillna(0)
56 
57  return df_rel_freqs

60 def compute_rel_AU_freq_cond_symm(df_counts_and, df_counts_or):
61  """!
62  Takes as input the absolute AU Co-Occurence frequencies stored in a Dataframe df_AU_freq and computes
63  for all AU combinations the symmetrized conditional probability for Co-Occurence:
64  P(AU_i = 1 \land AU_j = 1 | AU_i = 1 \lor AU_j = 1) = n_{AU_i \land AU_j} / n_{AU_i \lor AU_j}
65  The probability can be computed from the conditional co-occurrence frequencies by symmetrizing the matrix.
66 
67  @param df_AU_freq_rel_cond Pandas Multiindex Dataframe with conditional relative AU Co-Occurence frequencies; shape = (labels x nof_AU, nof_AU)
68  @retval df_AU_freq_rel_cond_symm Pandas Multiindex Dataframe with symmetrized conditional relative AU Co-Occurence frequencies; shape = (labels x nof_AU, nof_AU)
69  """
70 
71  # allocate memory for results
72  AU_names = list(df_counts_and)
73  df_rel_freqs = pd.DataFrame(0, index=AU_names, columns=AU_names)
74 
75  # Compute symmetrized conditional probability of Co-Occurance of AU Combinations:
76  for AUi in AU_names:
77  for AUj in AU_names:
78  df_rel_freqs.loc[AUi, AUj] = df_counts_and.loc[AUi, AUj] / df_counts_or.loc[AUi, AUj]
79 
80  return df_rel_freqs

83 def compute_adjacency_matrix(use_labels=[0,1,2,3,4,5,6,-1,-2,-3,-4,-5,-6], use_AUs=None, input_dir=None, AU_method="AUc", computation_method="symm", eps=None):
84  """!
85  Computes the weighted graph adjacency matrix based on co-occurrence counts obtained from a frequency analysis.
86 
87  @param use_labels (list, optional): A list of integers representing the labels to be used. Default is [0,1,2,3,4,5,6,-1,-2,-3,-4,-5,-6].
88  @param use_AUs (list, optional): A list of strings representing the Action Units (AUs) to use. Default is None.
89  @param input_dir (str, optional): Path to the directory containing the input files. Default is None.
90  @param AU_method (str, optional): Method used to obtain the Action Units. One of 'AUc' or 'AUr'. Default is 'AUc'.
91  @param computation_method (str, optional): Method used to compute the weights of the adjacency matrix. One of 'symm', 'cond' or 'uncond'. Default is 'symm'.
92  @param eps (float, optional): Threshold for filtering the adjacency matrix. Only top eps entries will be kept, rest will be set to 0. Default is None.
93 
94  @retval df_adj (DataFrame): A DataFrame representing the delta adjacency matrix.
95  """
96  # use_AUs select which AUs to use
97  # eps threshold: only take top eps entries, rest=0
98 
99  labels_valid = [0,1,2,3,4,5,6,-1,-2,-3,-4,-5,-6]
100  assert(np.isin(use_labels, labels_valid).all())
101  assert(AU_method in ["AUc", "AUr"]), "invalid AU_method provided. Valid is ['AUc', 'AUr]."
102  if not input_dir:
103  input_dir = os.path.join(os.path.dirname(os.path.abspath(__file__)), "results", "frequency_analysis")
104 
105  # load pair co-occurr counts and aggregate counts of use_labels
106  df_counts_and = pd.read_csv(os.path.join(input_dir, f"counts_{AU_method}_and.csv"), index_col=[0,1])
107  df_counts_and = df_counts_and[df_counts_and.index.get_level_values("label").isin(use_labels)]
108  df_counts_and = df_counts_and.groupby("AU").sum()
109 
110  # allocate memory for results
111  AUs = df_counts_and.columns
112  df_adj = pd.DataFrame(0, index=AUs, columns=AUs)
113 
114  # compute weights as probabilities approximated by relative frequencies
115  if computation_method == "symm":
116  df_counts_or = pd.read_csv(os.path.join(input_dir, f"counts_{AU_method}_or.csv"), index_col=[0,1])
117  df_counts_or = df_counts_or[df_counts_or.index.get_level_values("label").isin(use_labels)]
118  df_counts_or = df_counts_or.groupby("AU").sum()
119  df_adj = compute_rel_AU_freq_cond_symm(df_counts_and, df_counts_or)
120  elif computation_method == "cond":
121  df_adj = compute_rel_AU_freqs_cond(df_counts_and)
122  elif computation_method == "uncond":
123  nof_frames = pd.read_csv(os.path.join(input_dir, "nof_frames_per_label.csv"))
124  nof_frames = nof_frames[nof_frames.index.isin(use_labels)].sum()
125  df_adj = compute_rel_AU_freqs_uncond(df_counts_and, nof_frames)
126  else:
127  raise(ValueError("Invalid computation_method. Supported is [symm, cond, uncond]."))
128 
129  # filter for specified AUs/nodes
130  if use_AUs:
131  df_adj = df_adj.loc[use_AUs, use_AUs]
132 
133  # only keep top eps % entries, set other to 0
134  if eps:
135  assert (0 < eps < 1), "eps has to be in (0,1)"
136  percentile = np.percentile(df_adj.values.flatten(), eps*100)
137  df_adj[df_adj < percentile] = 0
138 
139  return df_adj

141 def compute_adjacency_matrix_delta(pain_labels=[3,-3, 6, -6], base_labels=[0], use_AUs=None, input_dir=None,
142  AU_method="AUc", computation_method="symm", eps=None):
143  """!
144  Compute the adjecency matrix as delta: adj_pain - adj_base given pain and base labels.
145 
146  @param pain_labels (list): A list of integers representing the pain labels. (default=[3,-3, 6, -6])
147  @param base_labels (list): A list of integers representing the base labels. (default=[0])
148  @param use_AUs (list): A list of integers representing the action units to use. (default=None)
149  @param input_dir (str): The input directory to use. (default=None)
150  @param AU_method (str): The method to use for the action units. (default="AUc")
151  @param computation_method (str): The method to use for the computation. (default="symm")
152  @param eps (float): A float value between 0 and 1 to use as the percentile value. (default=None)
153 
154  @retval df_adj (DataFrame): A DataFrame representing the delta adjacency matrix.
155  """
156  if not input_dir:
157  input_dir = os.path.join(os.path.dirname(os.path.abspath(__file__)), "results", "frequency_analysis")
158 
159  df_adj_pain = compute_adjacency_matrix(use_labels=pain_labels, use_AUs=use_AUs, input_dir=input_dir,
160  AU_method=AU_method, computation_method=computation_method)
161  df_adj_base = compute_adjacency_matrix(use_labels=base_labels, use_AUs=use_AUs, input_dir=input_dir,
162  AU_method=AU_method, computation_method=computation_method)
163  df_adj = (df_adj_pain - df_adj_base)
164  df_adj[df_adj<0] = 0
165 
166  if eps:
167  assert (0 < eps < 1), "eps has to be in (0,1)"
168  percentile = np.percentile(df_adj.values.flatten(), eps*100)
169  df_adj[df_adj < percentile] = 0
170 
171  return df_adj

174 def normalize_adj(df_adj):
175  """!
176  Normalizes the values of a given adjecency matrix (pd.DataFrame) to a range of [0, 1].
177 
178  @param df_adj (DataFrame) Adjacency matrix to be normalized.
179 
180  @retval df_adj (DataFrame) Normalized adjacency matrix.
181  """
182  df_adj = (df_adj - df_adj.min().min()) / (df_adj.max().max() - df_adj.min().min())
183  return df_adj
