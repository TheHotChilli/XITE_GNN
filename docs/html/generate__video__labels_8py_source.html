41 def generate_subject_video_labels(openface_dataset, subj_id, fs_video=25):
42  """!
43  Generates video labels for a subject.
44 
45  @param openface_dataset Instance of OpenFaceDataset class to load subject data
46  @param subj_id ID of the subject
47  @param fs_video Video frame rate (default 25 Hz)
48  """
49  try:
50  label_data = loadmat(os.path.join(dir_labels_raw, f"S{subj_id}.mat"))
51  except:
52  print(f"Warning: No label file available for S{subj_id}")
53  return
54  fs_label = label_data["fs"][0][0]
55  labels = np.squeeze(label_data["data"])
56  nof_samples = len(labels)
57 
58  # Bestimme wo bio label Intervalle beginnen und aufhören
59  idxs_end = np.diff(labels).nonzero()[0] # misses last endidx at nof_samples-1
60  idxs_start = idxs_end + 1 # misses first startidx at 0
61  interval_labels = np.concatenate([[labels[0]],labels[idxs_start]])
62 
63  # Rechne von der schnelleren bio sampling frequenz (~1000Hz) auf die langsamere video
64  # Frequenz (25 Hz) zurück. Nur jedes 40-e bio sample korrespondiert mit video sample.
65  # Frequenzverhältnis f_video/f_bio = 1/40 stellt diesen Zusammenhang her. Multilpliziere
66  # bio idxs mit 1/40 um video idxs zu erhalten. Achtung: Es ergeben sich krumme video idxs.
67  idxs_end = idxs_end * (fs_video/fs_label)
68  idxs_start = idxs_start * (fs_video/fs_label)
69 
70  # Runden der krummen video end idxs auf nächsten ganzen idx Wert
71  idxs_end_rounded = np.round(idxs_end)
72 
73  # Nehme diesen gerundeten end_idx als Referenz. Vergleiche nun ob die krummen start idxs oder
74  # end idxs näher an diesem Referenz Wert liegen.
75  is_endidx_mask = (np.abs(idxs_end_rounded - idxs_end) <= np.abs(idxs_end_rounded - idxs_start))
76 
77  idxs_end[is_endidx_mask] = idxs_end_rounded[is_endidx_mask] # An stellen wo krumme endidx näher nehme Referenz als endidx
78  idxs_end[~is_endidx_mask] = idxs_end_rounded[~is_endidx_mask] - 1 # An Stellen wo krumme startidx näher nehme Referenz-1 als endidx
79 
80  idxs_start[is_endidx_mask] = idxs_end_rounded[is_endidx_mask] + 1 # An Stellen wo krumme endidx näher nehme Referenz+1 als startidx
81  idxs_start[~is_endidx_mask] = idxs_end_rounded[~is_endidx_mask] # An Stellen wo krumme startidx näher nehme Referenz als startidx
82 
83  # add missing first start idx and last end idx
84  idxs_start = np.concatenate([[0], idxs_start])
85  idxs_end = np.append(idxs_end, np.round(nof_samples*(fs_video/fs_label)))
86 
87  # generate a video labels vector (label for each frame/sample)
88  #labels_video = np.ndarray(int(idxs_end[-1])+1, dtype=int)
89  nof_samples_video = len(openface_dataset.load_data(subj_id, target_channels=["AU01_r"]))
90  labels_video = np.empty(nof_samples_video); labels_video[:] = np.nan
91  for i in range(len(idxs_start)):
92  start_idx = int(idxs_start[i])
93  end_idx = int(idxs_end[i])
94  label = interval_labels[i]
95  labels_video[start_idx:end_idx+1] = label
96  # Falls labels länger recorded wurden
97  if end_idx >= (nof_samples_video-1):
98  break
99 
100  # write to file
101  if not os.path.exists(dir_video_labels):
102  os.makedirs(dir_video_labels)
103  path = os.path.join(dir_video_labels, f"S{subj_id}.csv")
104  pd.DataFrame(labels_video).to_csv(path, index=False)
105

107  if nof_samples_video > (idxs_end[-1]+1):
108  diff = nof_samples_video - idxs_end[-1]+1
109  print(f"Warning: For S{subj_id} video was recorded longer than labels (diff = {diff} samples = {diff*(1/fs_video):.2f}s; last available label = {interval_labels[-1]}).")
110  print(f"Generated video labelfile for S{subj_id}")
