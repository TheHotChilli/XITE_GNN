126 def run_training(cfg, path_out):
127  # print start
128  print("Starting training")
129 
130  # Dataset Inizialization
131  # adj_matrix = compute_adjacency_matrix_delta(pain_labels=[3,-3,6,-6], base_labels=[0])
132  adj_matrix = pd.read_csv(cfg["ADJ_PATH"], index_col=0).drop("AU28_c",axis=0).drop("AU28_c", axis=1)
133  dataset = OpenFace_Features_Dataset(root_path=cfg["DATA_PATH"], use_labels=cfg["PAIN_LABELS"]+cfg["BASE_LABLES"], adj_matrix=adj_matrix)
134  print("===========================")
135  print("Dataset:")
136  print(dataset.df.index.get_level_values("label").value_counts())
137 
138  # Model Initialization
139  assert (dataset.num_classes == 2), "Not a binary classification problem!"
140  in_channels = int(dataset.num_node_features)
141  hidden_channels = [int(round(multiplier * in_channels)) for multiplier in cfg["HIDDEN_CHANNELS"]]
142  module = importlib.import_module("XITE_GNN.training.BinaryClassification.models.{}".format(cfg["MODEL_TYPE"]))
143  model = getattr(module, cfg["MODEL_TYPE"])(in_channels=in_channels, hidden_channels=hidden_channels, out_channels=1, dropout=cfg["DROPOUT"], dropout_G=cfg["DROPOUT_G"])
144  loss_func = torch.nn.BCELoss()
145  optimizer = torch.optim.Adam(model.parameters(), lr=cfg["LR"], weight_decay=cfg["WEIGHT_DECAY"])
146 
147  # print model
148  print("===========================")
149  print("Model overview:")
150  print(model)
151 
152  # train and eval
153  print("===========================")
154  print("Starting {}-Fold Cross Validation...".format(cfg["NUM_FOLDS"]))
155  history, best_params = train.kfold_CV(dataset, model, cfg["DEVICE"], loss_func, optimizer, k=cfg["NUM_FOLDS"],
156  num_epochs=cfg["NUM_EPOCHS"], batch_size=cfg["BATCH_SIZE"], scheduler=None)
157 
158  # summary:
159  row_idx = ["test_acc_last", "test_acc_all", "train_acc_last", "train_acc_all",
160  "test_loss_last", "test_loss_all", "train_loss_last", "train_loss_all"]
161  df_summary = pd.DataFrame(0, index=row_idx, columns= ["val", "sd"])
162  for k in range(cfg["NUM_FOLDS"]):
163  df_summary.loc["test_acc_last","val"] += history[k]["test_acc"][-1] / cfg["NUM_FOLDS"]
164  df_summary.loc["train_acc_last","val"] += history[k]["train_acc"][-1] / cfg["NUM_FOLDS"]
165  df_summary.loc["test_acc_all","val"] += np.sum(history[k]["test_acc"]) / (cfg["NUM_FOLDS"] * cfg["NUM_EPOCHS"])
166  df_summary.loc["train_acc_all","val"] += np.sum(history[k]["train_acc"]) / (cfg["NUM_FOLDS"] * cfg["NUM_EPOCHS"])
167  df_summary.loc["test_loss_last","val"] += history[k]["test_loss"][-1] / cfg["NUM_FOLDS"]
168  df_summary.loc["train_loss_last","val"] += history[k]["train_loss"][-1] / cfg["NUM_FOLDS"]
169  df_summary.loc["test_loss_all","val"] += np.sum(history[k]["test_loss"]) / (cfg["NUM_FOLDS"] * cfg["NUM_EPOCHS"])

171

173  df_summary.loc["train_acc_last","sd"] += np.std([history[k]["train_acc"][-1] for k in range(cfg["NUM_FOLDS"])])
174  df_summary.loc["test_loss_last","sd"] += np.std([history[k]["test_loss"][-1] for k in range(cfg["NUM_FOLDS"])])
