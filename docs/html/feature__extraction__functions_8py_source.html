77 def signal_split_equal_part_mean(signal, sample_rate=1000, factor=10):
78  n = len(signal)
79  # a proposal to get an almost equal split with an appropriate size of each sub array
80  # same is done in the function signal_split_equal_part_std
81  # if we have a tonic stimuli which normally has a length of e.g. 60000 samples, then
82  # approx. 60 sub arrays are created
83  if n < sample_rate * factor:
84  t_i = n / sample_rate
85  else:
86  # to get also an almost equal number of sub arrays for phasic stimuli, the number of samples per sub array
87  # is adjusted by the factor, which leads to a 10th factor less samples per sub array
88  t_i = n / sample_rate * factor
89  split_array = np.array_split(signal, t_i)
90  mean_values_partitions = np.array([np.mean(x_i) for x_i in split_array])
91  return np.mean(mean_values_partitions)
