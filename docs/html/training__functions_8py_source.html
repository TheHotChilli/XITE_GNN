38 def train_epoch(model, dataloader, loss_func, optimizer, device, num_classes):
39  """!
40  Trains the input PyTorch model for a single epoch on the input data and returns the training loss and the confusion matrix.
41 
42  @param model PyTorch model to be trained.
43  @param dataloader PyTorch DataLoader object providing the data for training.
44  @param loss_func PyTorch loss function to be used during training.
45  @param optimizer PyTorch optimizer to be used during training.
46  @param device Device (CPU/GPU) on which to perform the computations.
47  @param num_classes Number of classes in the output of the model.
48  @retval Tuple A tuple containing the training loss (float) and the confusion matrix (torch.Tensor).
49  """
50  # set model to train mode (impacts behaviour of some layers like Dropout, BatchNorm etc)
51  model.train()
52  # initialize epoch loss and counter for correct samples
53  train_loss = 0
54  train_cf = 0
55  # enumerate train-samples/batches
56  for data in dataloader:
57  # copy tensor/graph/data to CUDA device
58  data = data.to(device)
59  # cleaer gradients
60  optimizer.zero_grad()
61  # perform forward path
62  y_hat = model(data) # batch_size x dim_out
63  # compute loss
64  loss = loss_func(y_hat, data.y.float().unsqueeze(1))
65  # perform backward path -> gradients
66  loss.backward()
67  # update model parameters
68  optimizer.step()
69  # aggregate evaluation metrics
70  train_loss += loss.item()
71  train_cf += confusion_matrix(data.y.detach().cpu(), \
72  y_hat.round().detach().cpu(), labels=list(range(num_classes)))
73  return (train_loss / len(dataloader)), train_cf

141  """!
142  Perform k-fold cross-validation on a dataset using a specified PyTorch model.
143

147  @param loss_func The loss function to use for training the model.

150  @param num_epochs The number of epochs to train the model for in each fold.
151  @param batch_size The batch size to use for training and evaluation.

154  """
155  subjects = dataset.subject_list
156  graph_data = dataset.to_torch_geometric()
157  # container for metrics per fold
158  history = {fold:{"train_loss":[], "train_acc":[], "train_cf":[], "train_subjects":[],

160  "train_loss_run": [], "train_acc_run":[], "train_cf_run":[],
161  "test_loss":[], "test_acc":[], "test_cf":[], "test_subjects":[]}
162  for fold in range(k)}

172  # print fold
